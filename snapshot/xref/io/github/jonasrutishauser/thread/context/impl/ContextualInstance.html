6   class ContextualInstance<T> {

8       private final CreationalContext<T> ctx;
9       private T instance;
10  
11      public ContextualInstance(Contextual<T> contextual, CreationalContext<T> ctx) {
12          this.contextual = contextual;
13          this.ctx = ctx;
14      }
15  
16      public T getInstance() {
17          if (instance == null) {
18              instance = contextual.create(ctx);
19          }
20          return instance;
21      }
22  
23      public void destroy() {

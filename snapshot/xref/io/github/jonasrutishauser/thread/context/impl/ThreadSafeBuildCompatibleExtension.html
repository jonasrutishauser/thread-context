26  @SkipIfPortableExtensionPresent(ThreadSafeExtension.class)
27  public class ThreadSafeBuildCompatibleExtension implements BuildCompatibleExtension {
28  
29      @Priority(LIBRARY_BEFORE + 500)
30      @Discovery
31      public void addContext(MetaAnnotations metaAnnotations) {
32          metaAnnotations.addContext(ThreadSafeScoped.class, true, ThreadSafeContext.class);
33      }
34  
35      @Priority(LIBRARY_AFTER + 900)
36      @Enhancement(types = Object.class, withSubtypes = true, withAnnotations = ThreadSafeScoped.class)
37      public void addInterceptor(ClassConfig classConfig) {
38          if (classConfig.info().hasAnnotation(ThreadSafeScoped.class)) {
39              classConfig.addAnnotation(ThreadSafeScopedInterceptor.class);
40          }
41      }
42  
43      @Registration(types = Object.class)
44      public void validateThreadSafeScopedBeans(BeanInfo bean, Messages messages) {
45          if ("io.github.jonasrutishauser.thread.context.ThreadSafeScoped".equals(bean.scope().name())
46                  && (bean.isProducerField() || bean.isProducerMethod())) {
47              messages.error("ThreadSafeScoped is not allowed on producer field or producer method in CDI Light", bean);
48          }
49      }
50  
51      @Priority(LIBRARY_BEFORE + 500)
52      @Synthesis
53      public void registerScopeShutdown(SyntheticComponents components, Types types) {
54          components.addObserver(Shutdown.class) //
55                  .priority(LIBRARY_AFTER + 900) //
56                  .observeWith(ScopeShutdown.class);
57      }
58  
59      public static class ScopeShutdown implements SyntheticObserver<Shutdown> {
60          @Override
61          public void observe(EventContext<Shutdown> event, Parameters params) throws Exception {
62              ((ThreadSafeContext) CDI.current().getBeanManager().getContext(ThreadSafeScoped.class)).shutdown();
63          }
64      }
